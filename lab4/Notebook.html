Uruchamiaj stream.start() uruchamiamy w osobnym demonie websocket ktry streamuje wyniki. Jeeli wystpi jaki bd po stronie front-endu (np. bd parsowania kolejnej linijki Pythona) fakt ten nie zostanie zgoszony do sparka i socket pozostanie otwarty! Naley pamita, by zamyka stream za kadym razem uywaj metody stop() (w powyszym przykadzie query.stop()). W przypadku utracenia referencji do zapytania, naley zastopowa ca sesj rwnie metod stop() (w powyszym przykadzie spark.stop())

from pyspark.sql.functions import col, window, avg

window_duration = "10 minutes"  # Total window size
slide_duration = "5 minutes"   # Sliding interval step size

# Transform the stream: Calculate average price in sliding windows grouped by product_id
result_df = stream.groupBy(
    window(col("time"), window_duration, slide_duration),  # Sliding time window
    col("product_id")  # Group by exchange relationship
).agg(
    avg("price").alias("avg_price")  # Average the `price` column
)

query = result_df.writeStream.format("console").outputMode("update").option("truncate", "false").start()

query.awaitTermination(20)  # Wait for 10 seconds to display results

from pyspark.sql import SparkSession
from pyspark.sql.functions import col, window, avg

spark = SparkSession.builder.config("spark.sql.streaming.schemaInference", True).config("spark.sql.streaming.forceDeleteTempCheckpointLocation", True).getOrCreate()  
stream = spark.readStream.format("ws").option("schema", "ticker").option("url", "ws://mock:8025").load()

# Define the parameters for window and watermark
window_duration = "10 seconds"
watermark_delay = "5 seconds"

# Process the data: apply watermark and aggregate
result_df = stream \
    .withWatermark("time", watermark_delay) \
    .groupBy(
        window(col("time"), window_duration),  # Fixed-size window of 10 seconds
        col("product_id")  # Group by product_id
    ) \
    .agg(
        avg("price").alias("avg_price")  # Calculate average price
    )

# Output the results to the console
query = result_df.writeStream \
    .format("console") \
    .outputMode("update") \
    .option("truncate", "false") \
    .start()

# Allow the query to run for 30 seconds
query.awaitTermination(30)
query.stop()

  • W pierwszej serii wysyane wiadomoci o znacznikach czasowych 0s, 14s, 7s
  • W drugiej serii wysyane s wiadomoci o znacznikach czasowych 15s, 8s, 21s - 8 jeszcze wlaczony, bo 5 sekund delaylapie go w tej grupie
  • W trzeciej serii wysyane s wiadomoci o znacznikach czasowych 4s, 17s - 4 jest zignorowane bo przekroczony delay 5 sekund

from pyspark.sql import SparkSession
from pyspark.sql.functions import col, window


spark = SparkSession.\
    builder.\
    config("spark.sql.streaming.schemaInference", True).\
    config("spark.sql.streaming.forceDeleteTempCheckpointLocation", True).\
    getOrCreate()  

stream = spark.\
    readStream.\
    format("ws").\
    option("schema", "ticker").\
    load()


# Filter for buy-side transactions
ticker_buy_stream = stream.filter(col("side") == "buy")

# Read `heartbeat` stream
heartbeat_stream = spark.readStream.format("ws").option("schema", "heartbeat").load()

# Define join window size
join_window_duration = "10 seconds"  # Join window size

# Add prefix 'ticker.' to all columns in ticker_buy_stream
ticker_buy_stream_prefixed = ticker_buy_stream.select(
    [col(c).alias(f"ticker_{c}") for c in ticker_buy_stream.columns]
)

# Add prefix 'heartbeat.' to all columns in heartbeat_stream
heartbeat_stream_prefixed = heartbeat_stream.select(
    [col(c).alias(f"heartbeat_{c}") for c in heartbeat_stream.columns]
)

# Perform the join
joined_stream = ticker_buy_stream_prefixed.join(
    heartbeat_stream_prefixed,
    on=col("ticker_trade_id") == col("heartbeat_last_trade_id"),  # Join on trade_id
    how="inner"  # Inner join
)
# Output joined stream to console
query = joined_stream.writeStream.format("console").outputMode("append").option("truncate", "false").start()

# Wait for termination
query.awaitTermination(1900)
query.stop()
